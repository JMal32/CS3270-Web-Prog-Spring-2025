Helper – deal N random tiles of any colour function dealRandomTiles(&$supply, $n = 6): array { $hand = []; for ($i=0; $i<$n; $i++) { $col=array_rand($supply); // pick a random colour key $hand[]=array_pop($supply[$col]); // remove from supply if (empty($supply[$col])) unset($supply[$col]); } return $hand; } // initial deal: six tiles to player 0 $players[0]['tiles']=dealRandomTiles($polypSupply, 6);
